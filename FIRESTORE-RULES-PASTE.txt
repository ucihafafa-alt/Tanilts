rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return signedIn()
        && exists(/databases/$(database)/documents/admins/$(request.auth.uid))
        && get(/databases/$(database)/documents/admins/$(request.auth.uid)).data.active == true;
    }

    function isOwner(uid) {
      return signedIn() && request.auth.uid == uid;
    }

    match /users/{uid} {
      allow create: if isOwner(uid)
        && request.resource.data.keys().hasAll([
          'phone', 'fullName', 'birthDate', 'age', 'gender', 'city',
          'maritalStatus', 'hasChildren', 'job', 'goal', 'preferredGender',
          'preferredAgeMin', 'preferredAgeMax', 'locationImportance',
          'acceptChildren', 'acceptPastMarriage', 'smoking', 'drinking',
          'relocate', 'about', 'dealBreakers', 'selfie',
          'verificationStatus', 'accountStatus', 'deletionRequested',
          'consentVersion', 'createdAt', 'updatedAt'
        ])
        && request.resource.data.keys().hasOnly([
          'phone', 'fullName', 'birthDate', 'age', 'gender', 'city',
          'maritalStatus', 'hasChildren', 'job', 'goal', 'preferredGender',
          'preferredAgeMin', 'preferredAgeMax', 'locationImportance',
          'acceptChildren', 'acceptPastMarriage', 'smoking', 'drinking',
          'relocate', 'about', 'dealBreakers', 'selfie',
          'verificationStatus', 'accountStatus', 'deletionRequested',
          'consentVersion', 'createdAt', 'updatedAt'
        ])
        && request.resource.data.phone is string
        && request.resource.data.phone.size() == 8
        && request.resource.data.fullName is string
        && request.resource.data.fullName.size() >= 2
        && request.resource.data.age is int
        && request.resource.data.age >= 18
        && request.resource.data.age <= 90
        && request.resource.data.preferredAgeMin is int
        && request.resource.data.preferredAgeMax is int
        && request.resource.data.preferredAgeMin >= 18
        && request.resource.data.preferredAgeMax <= 90
        && request.resource.data.preferredAgeMin <= request.resource.data.preferredAgeMax
        && request.resource.data.selfie is string
        && request.resource.data.selfie.size() > 1000
        && request.resource.data.selfie.size() < 750000
        && request.resource.data.verificationStatus == 'pending'
        && request.resource.data.accountStatus == 'active'
        && request.resource.data.deletionRequested == false
        && request.resource.data.createdAt is timestamp
        && request.resource.data.updatedAt is timestamp;

      allow get: if isOwner(uid) || isAdmin();
      allow list: if isAdmin();

      allow update: if isAdmin() || (
        isOwner(uid)
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          'fullName', 'city', 'job', 'preferredGender', 'preferredAgeMin',
          'preferredAgeMax', 'locationImportance', 'acceptChildren',
          'acceptPastMarriage', 'smoking', 'drinking', 'relocate', 'about',
          'dealBreakers', 'deletionRequested', 'updatedAt'
        ])
        && request.resource.data.verificationStatus == resource.data.verificationStatus
        && request.resource.data.accountStatus == resource.data.accountStatus
        && request.resource.data.phone == resource.data.phone
        && request.resource.data.selfie == resource.data.selfie
      );

      allow delete: if false;
    }

    match /appSettings/public {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /publicStats/app {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /matchRequests/{requestId} {
      allow create: if signedIn()
        && request.resource.data.userUid == request.auth.uid
        && request.resource.data.keys().hasAll([
          'userUid', 'type', 'serviceWindow', 'status', 'paymentStatus',
          'price', 'promoEligible', 'goal', 'preferredGender',
          'preferredAgeMin', 'preferredAgeMax', 'preferredLocations',
          'notes', 'paymentReference', 'cancelRequested',
          'createdAt', 'updatedAt'
        ])
        && request.resource.data.keys().hasOnly([
          'userUid', 'type', 'serviceWindow', 'status', 'paymentStatus',
          'price', 'promoEligible', 'goal', 'preferredGender',
          'preferredAgeMin', 'preferredAgeMax', 'preferredLocations',
          'notes', 'paymentReference', 'cancelRequested',
          'createdAt', 'updatedAt'
        ])
        && request.resource.data.type in ['regular', 'urgent']
        && request.resource.data.status == 'pending'
        && request.resource.data.paymentStatus in ['not_required', 'pending_review']
        && request.resource.data.price is int
        && request.resource.data.price >= 0
        && (
          (
            request.resource.data.type == 'urgent'
            && request.resource.data.promoEligible == false
            && request.resource.data.price == get(/databases/$(database)/documents/appSettings/public).data.urgentPrice
            && request.resource.data.paymentStatus == 'pending_review'
          )
          ||
          (
            request.resource.data.type == 'regular'
            && (
              (
                request.resource.data.promoEligible == true
                && request.resource.data.price == 0
                && request.resource.data.paymentStatus == 'not_required'
                && get(/databases/$(database)/documents/appSettings/public).data.promoOpen == true
                && get(/databases/$(database)/documents/publicStats/app).data.freeRegularUsed
                  < get(/databases/$(database)/documents/appSettings/public).data.promoLimit
              )
              ||
              (
                request.resource.data.promoEligible == false
                && request.resource.data.price == get(/databases/$(database)/documents/appSettings/public).data.regularPrice
                && request.resource.data.paymentStatus == 'pending_review'
              )
            )
          )
        )
        && request.resource.data.preferredAgeMin is int
        && request.resource.data.preferredAgeMax is int
        && request.resource.data.preferredAgeMin >= 18
        && request.resource.data.preferredAgeMax <= 90
        && request.resource.data.preferredAgeMin <= request.resource.data.preferredAgeMax
        && request.resource.data.cancelRequested == false
        && request.resource.data.createdAt is timestamp
        && request.resource.data.updatedAt is timestamp;

      allow read: if isAdmin()
        || (signedIn() && resource.data.userUid == request.auth.uid);

      allow update: if isAdmin() || (
        signedIn()
        && resource.data.userUid == request.auth.uid
        && request.resource.data.userUid == resource.data.userUid
        && request.resource.data.type == resource.data.type
        && request.resource.data.price == resource.data.price
        && request.resource.data.paymentStatus == resource.data.paymentStatus
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          'preferredAgeMin', 'preferredAgeMax', 'preferredLocations',
          'notes', 'status', 'cancelRequested', 'updatedAt'
        ])
        && (
          request.resource.data.status == resource.data.status
          || request.resource.data.status == 'cancel_requested'
        )
        && (
          request.resource.data.status != 'cancel_requested'
          || request.resource.data.cancelRequested == true
        )
      );

      allow delete: if false;
    }

    match /introductions/{introductionId} {
      function participant() {
        return signedIn()
          && resource.data.participants is list
          && request.auth.uid in resource.data.participants;
      }

      allow create: if isAdmin();
      allow read: if isAdmin() || participant();

      allow update: if isAdmin() || (
        participant()
        && (
          (
            request.auth.uid == resource.data.userAUid
            && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['responseA', 'updatedAt'])
            && request.resource.data.responseA in ['accepted', 'rejected']
          )
          ||
          (
            request.auth.uid == resource.data.userBUid
            && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['responseB', 'updatedAt'])
            && request.resource.data.responseB in ['accepted', 'rejected']
          )
        )
        && request.resource.data.participants == resource.data.participants
        && request.resource.data.userAUid == resource.data.userAUid
        && request.resource.data.userBUid == resource.data.userBUid
        && request.resource.data.status == resource.data.status
      );

      allow delete: if false;
    }

    match /notifications/{notificationId} {
      allow create: if isAdmin();
      allow read: if isAdmin()
        || (signedIn() && resource.data.userUid == request.auth.uid);
      allow update: if isAdmin() || (
        signedIn()
        && resource.data.userUid == request.auth.uid
        && request.resource.data.userUid == resource.data.userUid
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['read', 'readAt'])
      );
      allow delete: if false;
    }

    match /supportThreads/{uid} {
      allow create: if isOwner(uid)
        && request.resource.data.userUid == uid;
      allow read: if isOwner(uid) || isAdmin();
      allow update: if isAdmin() || (
        isOwner(uid)
        && request.resource.data.userUid == resource.data.userUid
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          'userName', 'userPhone', 'lastMessage', 'adminUnread',
          'userUnread', 'updatedAt', 'createdAt'
        ])
      );
      allow delete: if false;

      match /messages/{messageId} {
        allow read: if isOwner(uid) || isAdmin();
        allow create: if (
          isOwner(uid)
          && request.resource.data.senderType == 'user'
          && request.resource.data.senderUid == request.auth.uid
          && request.resource.data.text is string
          && request.resource.data.text.size() > 0
          && request.resource.data.text.size() <= 500
          && request.resource.data.createdAt is timestamp
        ) || (
          isAdmin()
          && request.resource.data.senderType == 'admin'
          && request.resource.data.senderUid == request.auth.uid
          && request.resource.data.text is string
          && request.resource.data.text.size() > 0
          && request.resource.data.text.size() <= 500
          && request.resource.data.createdAt is timestamp
        );
        allow update, delete: if false;
      }
    }

    match /pinResetRequests/{id} {
      allow create: if request.resource.data.phone is string
        && request.resource.data.phone.size() == 8
        && request.resource.data.status == 'pending'
        && request.resource.data.createdAt is timestamp;
      allow read, update: if isAdmin();
      allow delete: if false;
    }

    match /admins/{uid} {
      allow read: if isOwner(uid);
      allow write: if false;
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
